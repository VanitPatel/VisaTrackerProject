Storage Key Convention
// Pattern: {namespace}:{entity_type}:{id}[:{subkey}] // All keys: lowercase, colon-delimited, no spaces/slashes/quotes // Max key length: 200 chars | Max value size: 5MB // EXAMPLES: "users:profile:usr_alex_j_001" // User profile "cases:case:WAC2600234567" // Visa case "cases:timeline:WAC2600234567" // Case timeline array "docs:meta:doc_4521" // Document metadata "docs:ocr:doc_4521" // OCR extracted fields "notifs:list:usr_alex_j_001" // Notification list "notifs:prefs:usr_alex_j_001" // Notification preferences "billing:sub:sub_premium_001" // Subscription record "ai:pred:WAC2600234567" // AI prediction "ai:chat:usr_alex_j_001" // Chat history "chain:record:doc_4521" // Blockchain record (shared) "chain:vc:usr_alex_j_001" // Verifiable Credential (shared) "audit:event:evt_20260302_001" // Audit log event (shared) "cache:govt-api:USCIS:WAC2600234567" // API response cache "kafka:queue:case-status-events" // Kafka-backed event queue
